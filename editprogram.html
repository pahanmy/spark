Ralat Kebenaran Akses Pangkalan Data

Tindakan ini gagal kerana tetapan keselamatan (Security Rules) pada pangkalan data Firebase anda tidak membenarkannya. Ini BUKAN ralat pada kod aplikasi.

Untuk membetulkannya, anda MESTI mengemas kini fail firestore.rules di Firebase Console anda. 

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
  
    // --- TAMBAH/PASTIKAN RULE INI WUJUD ---
    match /artifacts/{appId}/public/data/spark_bookings/{bookingId} {
      allow read: if true;
      allow create: if request.auth != null;
    }
    // --------------------------------------

    match /artifacts/{appId}/public/data/spark_participants/{docId} {
       allow read: if true;
       allow create, update: if request.auth != null; 
    }
    match /artifacts/{appId}/public/data/spark_programs/{programId} {
      allow read: if true;
      allow create: if request.auth != null;
      allow update: if (request.auth != null && (request.resource.data.diff(resource.data).affectedKeys().hasOnly(['attendance']) || request.resource.data.diff(resource.data).affectedKeys().hasOnly(['surveyResponses']) || request.resource.data.affectedKeys().hasOnly(['evaluations']))) || (request.auth.uid == resource.data.penyelaras_uid);
      allow delete: if request.auth.uid == resource.data.penyelaras_uid;
    }
     match /artifacts/{appId}/public/data/spark_schools/{schoolId} {
      allow read: if true;
    }
    // Tambah rules lain jika perlu
  }
}
